async function loadAndShowLightning(mapIndex, startTimeStr, endTimeStr) { if (lightningVisible[mapIndex]) { hideLightning(mapIndex); return; } const btn = document.getElementById(`lightning-btn-${mapIndex}`); if (!btn) return; btn.disabled = true; btn.textContent = "Loading lightning..."; try { // ─── Parse the exact HOCO period ─── const start = parseHocoTime(startTimeStr); const end = parseHocoTime(endTimeStr || startTimeStr); if (isNaN(start.getTime()) || isNaN(end.getTime())) { throw new Error("Could not parse forecast time range"); } const startUnix = Math.floor(start.getTime() / 1000); const endUnix = Math.floor(end.getTime() / 1000); // Debug: show what time range we are looking for console.log(`Looking for lightning between ${start.toISOString()} → ${end.toISOString()}`); // ─── Fetch both files ─── let allStrikes = []; for (const url of [ 'https://raw.githubusercontent.com/HandryOutlook/lightning_data_new/refs/heads/main/lightning_data.json', 'https://raw.githubusercontent.com/HandryOutlook/lightning_data_new/refs/heads/main/lightning_data_2025_autumn.json' ]) { let jsonData = lightningCache.get(url); if (!jsonData || (Date.now() - jsonData.timestamp) > CACHE_TTL_MS) { const resp = await fetch(url, { cache: 'no-cache' }); if (!resp.ok) continue; jsonData = { data: await resp.json(), timestamp: Date.now() }; lightningCache.set(url, jsonData); } const raw = jsonData.data; const strikes = Array.isArray(raw) ? raw : (raw?.lightning_strikes || []); const filtered = strikes.filter(s => { if (!s.strike_time) return false; const t = Math.floor(new Date(s.strike_time).getTime() / 1000); return t >= startUnix && t <= endUnix; }); allStrikes.push(...filtered); } // ─── Deduplicate ─── const seen = new Map(); allStrikes = allStrikes.filter(s => { if (!s.coordinates || s.coordinates.length < 2) return false; const key = `${s.strike_time}|${s.coordinates[0].toFixed(4)}|${s.coordinates[1].toFixed(4)}`; if (seen.has(key)) return false; seen.set(key, true); return true; }); // ─── Final check ─── if (allStrikes.length === 0) { showToast(`No strikes in ${start.toLocaleDateString()} ${start.getHours()}:${start.getMinutes()} – ${end.toLocaleDateString()} ${end.getHours()}:${end.getMinutes()}`, "info"); console.log("Total strikes found before time filter:", allStrikes.length); // for debugging resetButton(btn); return; } // ─── Build GeoJSON & add layers (unchanged) ─── const geojson = { type: "FeatureCollection", features: allStrikes.map(s => ({ type: "Feature", geometry: { type: "Point", coordinates: [s.coordinates[0], s.coordinates[1]] }, properties: { time: s.strike_time, peak_current: null, cloud: "?" } })) }; const map = maps[mapIndex]; const sourceId = `lightning-src-${mapIndex}`; const layerId = `lightning-layer-${mapIndex}`; const clusterLayerId = `lightning-clusters-${mapIndex}`; [layerId, clusterLayerId].forEach(l => map.getLayer(l) && map.removeLayer(l)); if (map.getSource(sourceId)) map.removeSource(sourceId); map.addSource(sourceId, { type: 'geojson', data: geojson, cluster: allStrikes.length > 150, clusterMaxZoom: 14, clusterRadius: 38 }); if (allStrikes.length > 150) { map.addLayer({ id: clusterLayerId, type: 'circle', source: sourceId, filter: ['has', 'point_count'], paint: { 'circle-color': ['step', ['get', 'point_count'], '#8e44ad', 10, '#c0392b', 50, '#e74c3c'], 'circle-radius': ['step', ['get', 'point_count'], 18, 10, 26, 50, 34], 'circle-stroke-width': 1, 'circle-stroke-color': '#fff' }}); map.addLayer({ id: `${clusterLayerId}-count`, type: 'symbol', source: sourceId, filter: ['has', 'point_count'], layout: { 'text-field': '{point_count_abbreviated}', 'text-size': 14 }, paint: { 'text-color': '#fff' }}); } map.addLayer({ id: layerId, type: 'circle', source: sourceId, filter: allStrikes.length > 150 ? ['!', ['has', 'point_count']] : ['all'], paint: { 'circle-radius': 7, 'circle-color': '#d63031', 'circle-stroke-color': '#fff', 'circle-stroke-width': 1.5, 'circle-opacity': 0.9 } }, 'hoco-polygons'); lightningSources[mapIndex] = { sourceId, layerId, clusterLayerId, data: geojson }; lightningVisible[mapIndex] = true; btn.textContent = `⚡ Hide (${allStrikes.length})`; btn.style.background = '#c0392b'; btn.disabled = false; showToast(`✅ Showing ${allStrikes.length} lightning strikes`, "success"); } catch (err) { console.error("Lightning load error:", err); showToast("Failed to load lightning data", "error"); resetButton(btn); } }
